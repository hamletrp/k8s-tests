
Pre-requisites 
    Cluster provissioning permissions 
        ~/.aws/credentials 
        or a sts token 
        or a role setup
    To provisioned terraform resources

Create cluster 
    node group tags 
        k8s.io/cluster-autoscaler/enabled = true
        k8s.io/cluster-autoscaler/<your-cluster-name> = owned
    nodes labels 
        cluster-autoscaler-enabled=true

    eksctl create cluster -f ./gitops/eksctl-cluster-config.yml

    eksctl get cluster --name cluster-lab-10 --region us-east-1

Set storage classes
    k apply -f ./gitops/ebs-storageclass.yml

associate-iam-oidc-provider
    eksctl utils associate-iam-oidc-provider --region us-east-1  --cluster cluster-lab-10 --approve

get oicd provider 
    aws eks describe-cluster --region us-east-1 --name cluster-lab-10 --query "cluster.identity.oidc.issuer" --output text

copy oidc provider to tfvars

deploy terraform project 
    terraform apply --auto-approve

to update vpc-cni addon 
    eksctl update addon \
      --name vpc-cni \
      --cluster cluster-lab-10 \
      --region us-east-1 \
      --force

skip if addon serviceAccountRoleARN set on cluster config 
    patch serviceaccount/ebs-csi-controller-sa
